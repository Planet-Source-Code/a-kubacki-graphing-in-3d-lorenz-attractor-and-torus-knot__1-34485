counter = 0

For t = 0 To 2 * pi * p Step (2 * pi * p) / N

xb = xa
yb = ya
zb = za

X(counter) = xb
Y(counter) = yb
z(counter) = zb
asdf = counter

Next j

Private Sub rotate()
Dim i As Integer
Dim cx, sx, sy, cy, sz, cz As Double
Dim xt, yt, zt As Double
cx = Cos(xang)
sx = Sin(xang)
sy = Sin(yang)
cy = Cos(yang)
sz = Sin(zang)
cz = Cos(zang)

For i = 1 To asdf

yt = Y(i) * cx - z(i) * sx
zt = Y(i) * sx + z(i) * cx
Y(i) = yt
z(i) = zt
xt = X(i) * cy - z(i) * sy
zt = X(i) * sy + z(i) * cy
X(i) = xt
z(i) = zt
xt = X(i) * cz - Y(i) * sz
yt = X(i) * sz + Y(i) * cz
X(i) = xt
Y(i) = yt
Next i

End Sub
